function addProduct() {
    
  // Get the value from the element vendorName and store it into vendorName variable   
  let vendorName = document.getElementById("vendorName").value;
    
  // Get the value from the element name and store it into productName variable   
  let productName = document.getElementById("name").value;
   
  // Get the value from the element description and store it into productDescription variable      
  let productDescription = document.getElementById("description").value;
   
  // Get the value from the element image and store it into productImg variable  
  let productImg = document.getElementById("image").value;
   
  // Get the value from the element category and store it into productcategory variable  
  let productCategory = document.getElementById("category").value;
  
  // Get the value from the element actualPrice and store it into productActualPrice variable  
  let productActualPrice = document.getElementById("actualPrice").value;
  
  // Get the value from the element discount and store it into productDiscount variable  
  let productDiscount = document.getElementById("discount").value;
   
  // Get the value from the element discountedPrice and store it into productDiscountedPrice variable  
  let productDiscountedPrice = document.getElementById("discountedPrice").value;
    
  document.getElementById("error").innerHTML = "";
  document.getElementById("vendorName").style.borderColor = "#ced4da";
  document.getElementById("name").style.borderColor = "#ced4da";
  document.getElementById("description").style.borderColor = "#ced4da";
  document.getElementById("image").style.borderColor = "#ced4da";
  document.getElementById("category").style.borderColor = "#ced4da";
  document.getElementById("actualPrice").style.borderColor = "#ced4da";
  document.getElementById("discount").style.borderColor = "#ced4da";  
  document.getElementById("discountedPrice").style.borderColor = "#ced4da";    
     
    if (!vendorName) {
    document.getElementById("error").innerHTML = "Please enter vendor name";
    document.getElementById("vendorName").style.borderColor = "red";
  }
    else if (!productName) {
    document.getElementById("error").innerHTML = "Please enter product name";
    document.getElementById("name").style.borderColor = "red";
  } else if (!productDescription) {
    document.getElementById("error").innerHTML = "Please enter correct product details";
    document.getElementById("description").style.borderColor = "red";
  } else if (!productCategory) {
    document.getElementById("error").innerHTML = "Please select valid product category.";
    document.getElementById("category").style.borderColor = "red";
  } else if (!productImg) {
    document.getElementById("error").innerHTML = "Please enter an image url.";
    document.getElementById("image").style.borderColor = "red";
  } else if (!productActualPrice) {
    document.getElementById("error").innerHTML = "Please enter price of the product.";
    document.getElementById("actualPrice").style.borderColor = "red";
  } else if (!productDiscount) {
    document.getElementById("error").innerHTML = "Please enter product discount.";
    document.getElementById("discount").style.borderColor = "red";
  } 
  // Show error if discount is greater than 100% 
  else if (productDiscount>100) {
    document.getElementById("error").innerHTML = "Give discount less than 100 %";
    document.getElementById("discount").style.borderColor = "red";
  } 
  // Show error if ActualPrice is less than 1.
  else if (productActualPrice< 1) {
    document.getElementById("error").innerHTML = "Please enter price of the product.Price should be more than 0";
    document.getElementById("actualPrice").style.borderColor = "red";
  }
  // if none of the above error conditions are met then print all the form information using console.log()
  else {
      console.log(vendorName)
     console.log(productName)
     console.log(productDescription)
     console.log(productCategory)
     console.log(productActualPrice)
     console.log(productDiscount)
     console.log(productDiscountedPrice)
      
  }
}



function showFinalPrice() {
  productActualPrice = document.getElementById("actualPrice").value;
  productDiscount = document.getElementById("discount").value;

  productDiscountedPrice =
    productActualPrice - (productActualPrice * productDiscount) / 100;

  document.getElementById("discountedPrice").value = productDiscountedPrice;
}
